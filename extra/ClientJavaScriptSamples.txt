//create text to display and set its properties
let title = new StaticText();
title.name = "titleTxt";
title.fontSize = 24;
title.fontWeight = "bold";
title.foregroundColor = "#1F7A55";
title.text = "Visitor Data Logs";

//create a horizontal layout widget, set properties, and place it in the "FormVert" vertical layout on the Client page
let btnLO = new HorizontalLayout();
btnLO.name = "btnLO";
btnLO.heightPolicy = Widget.SizePolicy_SizeToParent;
btnLO.widthPolicy = Widget.SizePolicy_SizeToParent;
client.getWidget("FormVert").addChild(btnLO);


//create a button and set its properties, including an "onClick" event handler
let refreshBtn = new Button();
refreshBtn.name = "refreshBtn";
refreshBtn.buttonLabel = "Refresh Table";
refreshBtn.heightPolicy = Widget.SizePolicy_SizeToParent;
refreshBtn.widthPolicy = Widget.SizePolicy_SizeToParent;
refreshBtn.addEventHandler("onClick", () => {
    let tq = new TimedQueryParameters();
    tq.typeName = "com.acme.VisitorRecords_Small";
    tq.updateIntervalInSeconds = 10;
    
    client.data.tqds = client.createTimedQueryDataStream(tq);
    visTbl.dataStreamUUID = client.data.tqds.uuid;
    
    client.getWidget("btnLO").removeChild(refreshBtn);
});

//create a button and set some properties, including an "onClick" event handler that closes the Client
let closeBtn = client.clone(refreshBtn);
closeBtn.name = "closeBtn";
closeBtn.buttonLabel = "Close";
closeBtn.foregroundColor = "#1F7A55";
closeBtn.backgroundColor = "#FCF3BB";
closeBtn.addEventHandler("onClick", () => {
    client.terminate();
});


//create a datatable with columns entitled and typed, then create an event handler for when a row is clicked on
let visTbl = new DataTable();
visTbl.name = "visTbl";
visTbl.columnDescriptors = [
    {
        "propertyName" : "first_name",
        "title" : "First",
        "dataType" : "String"
    },
    {
        "propertyName" : "last_name",
        "title" : "Last",
        "dataType" : "String"
    }
];


visTbl.addEventHandler("onSelect", (client,page,extra) => {
    var parameters = {
        "record" : visTbl.selectedRowObject
    };
    client.popupPage("RowDets", "", parameters, () => {});
});

//add buttons to the bottom a layout widget
btnLO.addChild(refreshBtn, -1);
btnLO.addChild(closeBtn, -1);

//Add a datatable and text to the top of a layout widget
client.getWidget("FormVert").addChild(visTbl, 0);
client.getWidget("FormVert").addChild(title, 0);
